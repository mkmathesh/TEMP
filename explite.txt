package com.example.calc1;

import android.content.Intent;
import android.os.Bundle;
import android.widget.Button;
import androidx.appcompat.app.AppCompatActivity;

public class MainActivity extends AppCompatActivity {

    int score = 0;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main); // Q1 layout

        Button btnTrue = findViewById(R.id.btnTrue1);
        Button btnFalse = findViewById(R.id.btnFalse1);

        // Correct Answer = TRUE
        btnTrue.setOnClickListener(v -> {
            score = 1;
            goToQuestion2();
        });

        btnFalse.setOnClickListener(v -> {
            score = 0;
            goToQuestion2();
        });
    }

    private void goToQuestion2() {
        Intent intent = new Intent(MainActivity.this, Question2Activity.class);
        intent.putExtra("CURRENT_SCORE", score);
        startActivity(intent);
        finish(); // Prevent going back
    }
}

package com.example.calc1;

import android.content.Intent;
import android.os.Bundle;
import android.widget.Button;
import androidx.appcompat.app.AppCompatActivity;

public class Question2Activity extends AppCompatActivity {

    int runningScore;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main1); // Q2 layout

        // Get score from Q1
        runningScore = getIntent().getIntExtra("CURRENT_SCORE", 0);

        Button btnTrue = findViewById(R.id.btnTrue2);
        Button btnFalse = findViewById(R.id.btnFalse2);

        // Correct Answer = FALSE
        btnTrue.setOnClickListener(v -> goToResult(0));  // Wrong
        btnFalse.setOnClickListener(v -> goToResult(1)); // Correct
    }

    private void goToResult(int points) {
        runningScore += points;

        Intent intent = new Intent(Question2Activity.this, ResultActivity.class);
        intent.putExtra("FINAL_SCORE", runningScore);
        startActivity(intent);
        finish();
    }
}

package com.example.calc1;

import android.content.Intent;
import android.os.Bundle;
import android.widget.Button;
import android.widget.TextView;
import androidx.appcompat.app.AppCompatActivity;

public class ResultActivity extends AppCompatActivity {

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main2); // Result layout

        int finalScore = getIntent().getIntExtra("FINAL_SCORE", 0);

        TextView tvResult = findViewById(R.id.tvResult);
        tvResult.setText("Quiz Over! Total Score: " + finalScore + " / 2");

        Button btnRestart = findViewById(R.id.btnRestart);
        btnRestart.setOnClickListener(v -> {
            Intent intent = new Intent(ResultActivity.this, MainActivity.class);
            intent.setFlags(Intent.FLAG_ACTIVITY_CLEAR_TOP | Intent.FLAG_ACTIVITY_NEW_TASK);
            startActivity(intent);
            finish();
        });
    }
}
