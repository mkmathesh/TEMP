package com.example.quizapp;

import android.content.Intent;
import android.os.Bundle;
import android.widget.Button;
import androidx.appcompat.app.AppCompatActivity;

public class MainActivity extends AppCompatActivity {

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);

        Button btnJava = findViewById(R.id.btn_java);
        Button btnPython = findViewById(R.id.btn_python);

        // Explicit Intent to move to ResultActivity with data
        btnJava.setOnClickListener(v -> sendResult("Correct! Java is native."));
        btnPython.setOnClickListener(v -> sendResult("Wrong! Python is not native."));
    }

    private void sendResult(String message) {
        Intent intent = new Intent(MainActivity.this, ResultActivity.class);
        intent.putExtra("USER_SCORE", message); // Key-Value pair
        startActivity(intent);
    }
}
package com.example.quizapp;

import android.os.Bundle;
import android.widget.TextView;
import androidx.appcompat.app.AppCompatActivity;

public class ResultActivity extends AppCompatActivity {

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_result);

        TextView resultView = findViewById(R.id.result_text);

        // Get the Intent that started this activity and extract the string
        String message = getIntent().getStringExtra("USER_SCORE");
        
        if (message != null) {
            resultView.setText(message);
        }
    }
}
